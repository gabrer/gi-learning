149 SYMBOL* gi::dfa::table_filling(){
150  // The Table considered is only the upper triangular matrix, that can be saved in a linear array of size n(n-1)/2
151  // Conversion of index form matrix to array are:
152  //
153  // From linear index k, to (i,j) for tha matrix (i row, j column)
154  // i = n - 2 - floor(sqrt(-8*k + 4*n*(n-1)-7)/2.0 - 0.5)
155  // j = k + i + 1 - n*(n-1)/2 + (n-i)*((n-i)-1)/2
156  //
157  // From (i,j) to k
158  // Order such that i<j, then:
159  // k = (n*(n-1)/2) - (n-i)*((n-i)-1)/2 + j - i - 1
160  //
161  // check (http://stackoverflow.com/questions/27086195/linear-index-upper-triangular-matrix)
162 
163  // *** TABLE-FILLING ALGORITHM with witness ***
164  int i,j,k;
165  int n = num_states;
166  int tf_l = (num_states*(num_states-1))/2;
167 
168 
169  // Tabella delle coppie di stati distinti
170  SYMBOL* distinti = new SYMBOL[tf_l];
171 
172  // Inizializzo
173  //for(k=0; k<tf_l; ++k)
174  //distinti[k]= tableFillingState::TF_STATE_N;
175 
176  // Distinguo tra accettanti e non accettanti
177  for(i=0; i<(num_states-1); ++i)
178  for(j=i+1; j<num_states; ++j){
179  // Se uno è accettante mentre l'altro no
180  k= (n*(n-1)/2) - (n-i)*((n-i)-1)/2 + j - i - 1;
181  if(ttable[i][dim_alphabet] != ttable[j][dim_alphabet]){
182  distinti[k] = DFA_TF_STATE_X;
183  }else
184  distinti[k]= DFA_TF_STATE_N;
185  }
186 
187 
188  // Loop minimizzante
189  // Tiene traccia se la tabella è stata modificata durante l'ultimo ciclo
190  bool modificata = true;
191  while(modificata)
192  {
193  modificata = false;
194  int stato_arrivo_1;
195  int stato_arrivo_2;
196 
197  for(i=0; i<(num_states-1); ++i){
198  for(j=i+1; j<num_states; ++j){
199  k= (n*(n-1)/2) - (n-i)*((n-i)-1)/2 + j - i - 1;
200  //V cout << endl << "SP1: "<<i << ", SP2: "<<j;
201  if(distinti[k] == DFA_TF_STATE_N){
202 
203  for(SYMBOL w=0; w<dim_alphabet; ++w)
204  {
205  stato_arrivo_1 = ttable[i][w];
206  stato_arrivo_2 = ttable[j][w];
207 
208  // Lo stato di arrivo letto nel dfa potrebbe essere una coppia del tipo (2,2)
209  if(stato_arrivo_1 == stato_arrivo_2)
210  continue;
211 
212  // Facendo la tabella delle coppie di stati visitati noto che sempre j>i
213  // quindi le coppie devono avere sempre i<j
214  if(stato_arrivo_2 < stato_arrivo_1){
215  int tmp = stato_arrivo_1;
216  stato_arrivo_1 = stato_arrivo_2;
217  stato_arrivo_2 = tmp;
218  }
219 
220  // Se finisco su me stesso vado avanti
221  if(stato_arrivo_1 == i && stato_arrivo_2 == j)
222  continue;
223 
224  //V cout << endl <<"SA1: "<<stato_arrivo_1 << ", SA2: "<<stato_arrivo_2 << " --> " << (int)distinti[stato_arrivo_1][stato_arrivo_2] << endl;
225  // Se la coppia di arrivo è distinta, lo è anche quella originaria
226  int i1 = stato_arrivo_1, j1 = stato_arrivo_2;

228 
229  if(distinti[k1] != DFA_TF_STATE_N){
230  //TODO: è corretto?
231  //distinti[k1] = w + '0'; //scrive l'indice (in char) della lettera dell'alfabeto per cui i due stati differiscono
232  //distinti[k] = tableFillingState::TF_STATE_O;
233  distinti[k] = w;
234  modificata = true;
235  break; // Necessario!
236  }
237  }
238  }
239  }
240  }
241  }

244  #ifdef DEBUG_2
245  cout << "--------------------------" << endl;
246  cout << "Tabella dei distinti " << endl;
247  cout << "--------------------------" << endl;
